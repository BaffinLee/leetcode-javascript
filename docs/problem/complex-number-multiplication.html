    Problem

    Given two strings representing two complex numbers.

    You need to return a string representing their multiplication. Note i2 = -1 according to the definition.

    Example 1:

    Input: "1+1i", "1+1i"
Output: "0+2i"
Explanation: (1 + i) * (1 + i) = 1 + i2 + 2 * i = 2i, and you need convert it to the form of 0+2i.

    Example 2:

    Input: "1+-1i", "1+-1i"
Output: "0+-2i"
Explanation: (1 - i) * (1 - i) = 1 + i2 - 2 * i = -2i, and you need convert it to the form of 0+-2i.

    Note:

    • The input strings will not have extra blank.
    • The input strings will be given in the form of a+bi, where the integer a and b will both belong to the range of [-100, 100]. And the output should be also in this form.

    Solution 1

    /**
 * @param {string} a
 * @param {string} b
 * @return {string}
 */
var complexNumberMultiply = function(a, b) {
  var a = parse(a);
  var b = parse(b);
  var x = (a.x * b.x) - (a.y * b.y);
  var y = (a.x * b.y) + (a.y * b.x);
  return stringify(x, y);
};

var parse = function (str) {
  var res = /^(-?\d+)\+(-?\d+)i$/.exec(str);
  return { x: Number(res[1]), y: Number(res[2]) };
};

var stringify = function (x, y) {
  return `${x}+${y}i`
};

    /**
 * @param {string} a
 * @param {string} b
 * @return {string}
 */
var complexNumberMultiply = function(a, b) {
  var a = parse(a);
  var b = parse(b);
  var x = (a.x * b.x) - (a.y * b.y);
  var y = (a.x * b.y) + (a.y * b.x);
  return stringify(x, y);
};

var parse = function (str) {
  var index = str.indexOf('+');
  var x = Number(str.substring(0, index));
  var y = Number(str.substring(index + 1, str.length - 1));
  return { x: x, y: y };
};

var stringify = function (x, y) {
  return `${x}+${y}i`
}
