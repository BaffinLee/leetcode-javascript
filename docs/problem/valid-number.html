  • @return {boolean} */ var isNumber = function(s) { var state = [ {}, {'blank': 1, 'sign': 2, 'digit':3, '.':4}, {'digit':3, '.':4}, {'digit':3, '.':5, 'e':6, 'blank':9}, {'digit':5}, {'digit':5, 'e':6, 'blank':9}, {'sign':7, 'digit':8}, {'digit':8}, {'digit':8, 'blank':9}, {'blank':9} ]; var validState = [3, 5, 8, 9]; var currentState = 1; var len = s.length; var str = ''; var type = ''; for (var i = 0; i < len; i++) { str = s[i]; if (str >= '0' && str <= '9') { type = 'digit'; } else if (str === '+' || str === '-') { type = 'sign'; } else if (str === ' ') { type = 'blank'; } else { type = str; } if (state[currentState][type] === undefined) { return false; } else { currentState = state[currentState][type]; } } if (validState.indexOf(currentState) === -1) { return false; } else { return true; } }; ```
